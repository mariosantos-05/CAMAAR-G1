def require_admin
  unless admin?
    flash[:alert] = "Acesso negado. Apenas administradores."

    if logged_in?
      # Se é aluno tentando acessar área admin, manda pro dashboard de aluno
      redirect_to avaliacoes_path
    else
      # Se não tá logado, manda pro login
      redirect_to root_path
    end
  end
end
