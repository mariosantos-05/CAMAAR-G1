def call
  json_data = parse_json_file

  ActiveRecord::Base.transaction do
    json_data.each_with_index do |entry, index|

# Lê e valida o formato JSON do arquivo.
#
# Args: Nenhum.
# Returns: Array de Hashes.
# Side Effects: Leitura de disco.
def parse_json_file
  content = File.read(@file_path)
  data = JSON.parse(content)

  raise InvalidFileError, &quot;O JSON deve ser uma lista (Array) de objetos.&quot; unless data.is_a?(Array)
  data
rescue JSON::ParserError
  raise InvalidFileError, &quot;O arquivo não é um JSON válido.&quot;
end
